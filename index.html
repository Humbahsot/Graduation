chicken : '—', Side: attending === 'yes' ? side : '—', Drink: attending === 'yes' ? drink : '—', Dietary: dietary || '—' }; await tableClient.upsertEntity(rsvpEntry, "Replace"); context.res = { status: 201, headers: { "Content-Type": "application/json" }, body: { message: "RSVP logged successfully.", registeredName: name } }; } catch (error) { context.log.error("Error upserting entity to Table Storage:", error); context.res = { status: 500, body: "Failed to store registration." }; } } };
